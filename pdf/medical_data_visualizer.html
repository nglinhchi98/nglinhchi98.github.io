#Define the function to draw the catplot
def draw_cat_plot():
    # Melt the dataframe to reformat the categorical variables
    df_cat = pd.melt(df,
                     id_vars='cardio',
                     value_vars=['cholesterol', 'gluc', 'smoke', 'alco', 'active', 'overweight'],
                     var_name='variable',
                     value_name='value')

    # Group by cardio, variable, and value, and count occurrences
    df_cat = df_cat.groupby(['cardio', 'variable', 'value']).size().reset_index(name='total')

    # Create the catplot
    g = sns.catplot(data=df_cat,
                    x='variable',
                    y='total',
                    col='cardio',  # Separate plots for cardio=0 and cardio=1
                    hue='value',   # Color by value
                    kind='bar')    # Bar plot

    # Adjust layout to prevent overlap
    g.fig.tight_layout()

    # Explicitly return the figure to ensure it displays
    return g.fig

# Call the function and display the plot
fig = draw_cat_plot()
plt.show()

# Draw Heat Map
def draw_heat_map():
    # Clean the data
    df_heat = df[(df['ap_lo'] <= df['ap_hi']) |
                 (df['height'] >= df['height'].quantile(0.025)) |
                 (df['height'] > df['height'].quantile(0.975)) |
                 (df['weight'] < df['weight'].quantile(0.025)) |
                 (df['weight'] > df['weight'].quantile(0.975))]
    # Calculate the correlation matrix
    corr = df_heat.corr()
    # Generate a mask for the upper triangle
    mask = np.triu(np.ones_like(df_heat.corr()))
    # Set up the matplotlib figure
    fig, ax = plt.subplots(figsize=(10,8))
    # Draw the heatmap with 'sns.heatmap()'
    sns.heatmap(corr, 
                annot=True,  # Show correlation values
                mask=mask,   # Mask upper triangle
                fmt='.2f',   # Format numbers to 2 decimal places
                cmap='coolwarm',  # Color scheme
                vmin=-1, vmax=1,  # Correlation range
                center=0,         # Center the colormap at 0
                square=True,      # Make cells square
                ax=ax)            # Use the specified axis
    
    # Adjust layout to prevent cutoff
    plt.tight_layout()
    
    # Do not modify the next two lines
    fig.savefig('heatmap.png')
    return fig

# Call the function to draw the heatmap
fig = draw_heat_map()

# Display the plot in Jupyter Notebook
plt.show()
